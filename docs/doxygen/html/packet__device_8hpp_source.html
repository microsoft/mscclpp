71  MSCCLPP_DEVICE_INLINE bool readOnce(uint32_t flag, uint2& data) const {
72 #if defined(MSCCLPP_DEVICE_CUDA)
73  uint32_t flag1, flag2;
74  asm volatile("ld.volatile.global.v4.u32 {%0,%1,%2,%3}, [%4];"
75  : "=r"(data.x), "=r"(flag1), "=r"(data.y), "=r"(flag2)
76  : "l"(&raw_));
77  return (flag1 != flag) || (flag2 != flag);
78 #else // !defined(MSCCLPP_DEVICE_CUDA)
79  ulonglong2 reg;
80  reg.x = atomicLoad(&(raw_.x), memoryOrderRelaxed);
81  reg.y = atomicLoad(&(raw_.y), memoryOrderRelaxed);
82  uint4* ptr = reinterpret_cast<uint4*>(&reg);
83  data.x = ptr->x;
84  data.y = ptr->z;
85  return (ptr->y != flag) || (ptr->w != flag);
86 #endif
87  }
